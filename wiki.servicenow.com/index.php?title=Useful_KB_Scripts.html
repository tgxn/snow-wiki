Name: Group Separation in Knowledge Base

Type: Business Rule

Table: kb_knowledge

Description: Create a Group reference field on the KB article form. Create a "before query" business rule on the kb_knowledge table with the following script. Then users will see all knowledge base articles with no group, and also those articles with a group of which they're a member.

Parameters:

Script: 

if (!gs.hasRole("admin") && gs.getSession().isInteractive()) {
   var qc1 = current.addQuery('u_group',"");
   var gra = new GlideRecord('sys_user_grmember');
   gra.addQuery('user', gs.getUserID());
   gra.query();
   while (gra.next()) {
      qc1.addOrCondition('u_group', gra.group);
   }
}

Alternatively, use this script for groups with parent/child relationships: 

if (!gs.hasRole("admin") && gs.getSession().isInteractive()) {
   var qc1 = current.addQuery('u_group',"");
   qc1.addOrCondition('u_group', "javascript:getMyGroups()");
}

Name: Retire Expired Knowledge Base Articles

Type: Scheduled Script

Table:

Description: Use as a daily scheduled script to retire expired knowledge base articles.

Parameters:

Script: 

 doit();

function doit() { 

  var gr = new GlideRecord('kb_knowledge');
 gr.addQuery('valid_to','&lt;',gs.now());
 gr.addQuery('workflow_state','!=','retired');
 gr.query();
 while (gr.next()) {
   gr.workflow_state = 'retired';
   gs.print('KB article ' + gr.number + ' retired');
   gr.update();
 }

}
