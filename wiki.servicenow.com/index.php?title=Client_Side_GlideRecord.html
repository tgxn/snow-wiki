1 Overview

Client-side GlideRecord enables the use of some GlideRecord functionality inside client-side scripts, such as client scripts and UI policy scripts.

The client-side API is similar to that of server-side GlideRecord, with the following differences:

  • Client-side GlideRecord has a limited set of functionality.
  • Queries made with the client-side GlideRecord are executed on the server. Therefore, a request must be made from the client browser to obtain the record data. Precautions should be made to ensure that queries are optimized and that they do not run synchronously.
  • Client-side GlideRecord does not support Dot-Walking.

For more about using client-side GlideRecord, see the Using GlideRecord query from client section in What are some of the available helper functions?.

Client-side GlideRecord is not supported in scoped applications. Instead, create a script include and use GlideAjax, or use the REST APIs.

1.1 Where To Use

Client-side GlideRecord methods can be called from any client-side script in a global application; however, other, more efficient, methods are recommended and should be investigated first. For more information, see Client Script Best Practices.

setLimit(maxQuery) Sets the limit for how many records will be returned by the GlideRecord query.
getLimit() Gets the limit of records to be returned by the GlideRecord query.


3 Method Detail

3.1 addQuery(fieldName, operator, value)

Adds a filter to return records based on 1, 2, or 3 arguments.

  • 1 argument adds an encoded query string.
  • 2 arguments return records where the field is equal to the value (or is in a list of values).
  • 3 arguments return records where the field meets the specified condition (field, operator, and value).

This method can accept an IN operator (see example 2).

3.1.1 Input Fields

Parameters:

  • fieldName - (String) Name of the field.
  • operator - (Object) An operator for the query.
  • value - (Object) Value to query on.

3.1.2 Output Fields

Returns: void

3.1.3 Example 1

var rec = new GlideRecord('incident');
rec.addQuery('active', true);
rec.addQuery('sys_created_on', ">", "2010-01-19 04:05:00");
rec.query(recResponse);

function recResponse(rec) {
	while (rec.next()) { 
		rec.active = false;
		alert('Active incident ' + rec.number + ' closed');
		rec.update();
	} 
}

3.1.4 Example 2 - Using the IN Operator

var que = new GlideRecord('incident');
que.addQuery('number','IN','INC00001,INC00002');
que.query(queResponse);

function queResponse(que) {
	while(que.next()) {
		//do something....
	} 
}

3.1.5 Controlling Invalid Queries

By default, queries with invalid field names run but ignore the invalid condition. For more strict query control in client scripts, use GlideAjax and make the call on the server.


3.1.5.1 Example

/*
 * Default behavior if no property or property = false
 * Remove invalid query condition
 */
var gr = new GlideRecord("incident");
gr.addQuery("field_that_doesnt_exist", "my value");
gr.addQuery("active", true);
gr.query(grResponse);

function grResponse(gr) {
  while(gr.next()) {
		//do something....
  } 
}
//returns records matching active=true

/*
 * Property = true
 * Invalidate entire query if invalid condition exists
 */
var gr = new GlideRecord("incident");
gr.addQuery("field_that_doesnt_exist", "my value");
gr.addQuery("active", true);
gr.query(grResponse);

function grResponse(gr) {
  while(gr.next()) {
		//do something....
  } 
}
// returns no records
//also generated log message "field name 'field_that_doesnt_exist' not found in table 'incident'"


/*
 * Property = true, override strict query for session
 * Same behavior as setting property = false except only applies to this session
 */

var gr = new GlideRecord("incident");
gr.addQuery("field_that_doesnt_exist", "my value");
gr.addQuery("active", true);
gr.query(grResponse);

function grResponse(gr) {
  while(gr.next()) {
		//do something....
  } 
}
//returns records matching active=true


3.2 getEncodedQuery()

Gets the encoded query as a string.

3.2.1 Input Fields

Parameters:

  • None

3.2.2 Output Fields

Returns: (String) the encoded query.

3.3 deleteRecord(responseFunction)

Deletes a single record.

3.3.1 Input Fields

Parameters:

  • responseFunction - (String) The response function for the Ajax callback.

3.3.2 Output Fields

Returns: (boolean) true if record was deleted or false if none found to delete.

3.3.3 Example

var rec = new GlideRecord('incident');
rec.addQuery('active',false);
rec.query(recResponse);

function recResponse(rec) {
   while (rec.next()) { 
      alert('Inactive incident ' + rec.number + ' deleted');
      rec.deleteRecord();
   }
}


3.4 get(sys_id)

Executes a GlideRecord query for a record with the specified sys_id. This method is expected to be used to query for single records, so a next operation on the recordset is performed by this method before returning.

3.4.1 Input Fields

Parameters:

  • sys_id - (Object) The sys_id to query.

3.4.2 Output Fields

Returns: (boolean) true if one or more matching records were found, or false if no matches were found.

3.5 Example

function kbWriteComment(id, comments) {
 var fb = new GlideRecord('kb_feedback');
 fb.initialize();
 fb.article = id;
 fb.comments = comments;
 fb.insert();
}

function kbGetText(id) {
 var gr = new GlideRecord('kb_knowledge');
 if (gr.get(id)) {
 gr.u_times_used = gr.u_times_used + 1;
 gr.u_last_used = gs.nowDateTime();
 gr.update();
 // return "Knowledge article " + gr.number + ":\n" +
 // "[code]" + gr.text + "[/code]";
 return gr.number;
 }


3.6 getTableName()

Retrieves the table name associated with this GlideRecord.

3.6.1 Input Fields

Parameters:

  • None

3.6.2 Output Fields

Returns: (String) the table name.

3.6.3 Example

var item = new GlideRecord('sc_request'); 
item.addQuery('sys_id', current.request); 
item.query(itemResponse); 

function itemResponse(item) {
   alert('The table is ' + item.getTableName());
}


3.7 hasNext()

Determines if there are any more records in the GlideRecord.

3.7.1 Input Fields

Parameters:

  • None

3.7.2 Output Fields

Returns: (boolean) true if there are more records in the query set.

3.7.3 Example

if (gr.hasNext()) {
  dothis(); // found it, do it
} else {
  dothat(); // didn't find it
}


3.8 insert(responseFunction)

Inserts a new record using the field values that have been set for the current record.

3.8.1 Input Fields

Parameters:

  • responseFunction - (String) The response function for the Ajax callback.

3.8.2 Output Fields

Returns: (String) the unique ID of the inserted record, or null if the record is not inserted.

3.8.3 Example

var gr = new GlideRecord('to_do');
gr.initialize(); 
gr.name = 'first to do item'; 
gr.description = 'learn about GlideRecord'; 
gr.insert();


3.9 next()

Moves to the next record in the GlideRecord.

3.9.1 Input Fields

Parameters:

  • None

3.9.2 Output Fields

Returns: (boolean) false if there are no more records in the query set.

3.9.3 Example

var rec = new GlideRecord('incident');
rec.query(recResponse);

function recResponse(rec) {
  while (rec.next()) { 
    alert(rec.number + ' exists');
  }
}


3.10 addOrderBy(column)

Adds a column to order by in the query.

3.10.1 Input Fields

Parameters:

  • f - (String) The column to add.

3.10.2 Output Fields

Returns: void


3.11 orderBy(f)

Specifies an orderBy column (this may be called more than once to order by multiple columns).

3.11.1 Input Fields

Parameters:

  • f - (String) The column name used to order the recordset.

3.11.2 Output Fields

Returns: void

3.11.3 Example

function UpdateProjectWBS(project) {
  var count = 0;
  var child = new GlideRecord('pm_project_task');
  child.addQuery('parent', project.sys_id);
  child.orderBy('order');
  child.orderBy('number');
  child.query(childResponse);
  g_form.addInfoMessage(count + ' Project Tasks updated');
}

function childResponse(child) {
  var len = child.getRowCount().toString().length;
  var seq = 0;
  while (child.next()) {
    count += UpdateProjectTaskWBS(child, 1, ++seq, len, '');
  }
}


3.12 query(responseFunction)

Runs the query against the table based on the addQuery() filter. This queries the GlideRecord table as well as any references of the table.

3.12.1 Input Fields

Parameters:

  • responseFunction - (String) The response function for the Ajax callback.

3.12.2 Output Fields

Returns: void

3.12.3 Example

var rec = new GlideRecord('incident');
rec.query(recResponse);

function recResponse(rec) {
  while (rec.next()) { 
   alert(rec.number + ' exists');
  } 
}


3.13 setLimit(maxQuery)

Sets the limit for how many records are in the GlideRecord.

3.13.1 Input Fields

Parameters:

  • maxQuery - (int) The limit for number of records to fetch.

3.13.2 Output Fields

Returns: void

3.13.3 Example

var gr = new GlideRecord('incident');
gr.orderByDesc('sys_created_on');
gr.setLimit(10);
gr.query(grResponse);

function grResponse(gr) {
  while (gr.next()) { 
     //do something...
  } 
}


3.14 getLimit()

Returns the limit for records to be returned by the GlideRecord query.

3.14.1 Input Fields

Parameters:

  • None

3.14.2 Output Fields

Returns: (int) the limit for records to be returned by the GlideRecord query.
