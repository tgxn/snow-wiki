1 Overview

Level Platforms is a monitoring tool that sends alerts when a server is down.  This Web Services integration creates incidents in ServiceNow when an alert is created in Level Platforms, and ServiceNow updates Level Platforms when the incident is closed.

2 Implementation

2.1 Inbound Alerts to create Incident

To create incidents from alerts in Level Platforms, LPI use a static WSDL to define the web service it will consume. Copy the following WSDL and create a new Static WSDL record with the name "incident_create". System Web Services -> Static WSDL -> New 

<?xml version="1.0" encoding="UTF-8"?>
<wsdl:definitions xmlns:soap="http://schemas.xmlsoap.org/wsdl/soap/" 
  xmlns:tns="http://www.service-now.com/incident" xmlns:wsdl="http://schemas.xmlsoap.org/wsdl/" 
  xmlns:xsd="http://www.w3.org/2001/XMLSchema" 
  name="NewWSDLFile" 
  targetNamespace="http://www.service-now.com/incident">
  <wsdl:types>
    <xsd:schema targetNamespace="http://www.service-now.com/incident">
      <xsd:element name="Create">
        <xsd:complexType>
          <xsd:sequence>
            <xsd:element name="in" type="xsd:string"/>
          </xsd:sequence>
        </xsd:complexType>
      </xsd:element>
      <xsd:element name="CreateResponse">
        <xsd:complexType>
          <xsd:sequence>
            <xsd:element name="out" type="xsd:string"/>
          </xsd:sequence>
        </xsd:complexType>
      </xsd:element>
    </xsd:schema>
  </wsdl:types>
  <wsdl:message name="CreateRequest">
    <wsdl:part name="alert_category" type="xsd:string"/>
    <wsdl:part name="alert_details" type="xsd:string"></wsdl:part>
    <wsdl:part name="alert_id" type="xsd:string"></wsdl:part>
    <wsdl:part name="alert_severity" type="xsd:string"></wsdl:part>
    <wsdl:part name="alert_status" type="xsd:string"></wsdl:part>
    <wsdl:part name="alert_title" type="xsd:string"></wsdl:part>
    <wsdl:part name="technician_assigned" type="xsd:string"></wsdl:part>
    <wsdl:part name="ticket_priority" type="xsd:string"></wsdl:part>
    <wsdl:part name="device_id" type="xsd:string"></wsdl:part>
    <wsdl:part name="device_name" type="xsd:string"></wsdl:part>
    <wsdl:part name="site_id" type="xsd:string"></wsdl:part>
    <wsdl:part name="site_name" type="xsd:string"></wsdl:part>
    <wsdl:part name="site_contact" type="xsd:string"></wsdl:part>
    <wsdl:part name="site_phone" type="xsd:string"></wsdl:part>
    <wsdl:part name="site_email" type="xsd:string"></wsdl:part>
    <wsdl:part name="alert_url" type="xsd:string"></wsdl:part>
    <wsdl:part name="alert_guid" type="xsd:string"></wsdl:part>
  </wsdl:message>
  <wsdl:message name="CreateResponse">
    <wsdl:part name="output" type="xsd:string"></wsdl:part>
  </wsdl:message>
  <wsdl:portType name="Incident">
    <wsdl:operation name="Create">
      <wsdl:input message="tns:CreateRequest" name="IncidentInput"/>
      <wsdl:output message="tns:CreateResponse" name="IncidentOutput"/>
    </wsdl:operation>
  </wsdl:portType>
    <wsdl:binding name="IncidentSOAP" type="tns:Incident">
      <soap:binding style="rpc"
	transport="http://schemas.xmlsoap.org/soap/http" />
        <wsdl:operation name="Create">
	<soap:operation
	  soapAction="http://www.service-now.com/incident/Create" />
	  <wsdl:input name="IncidentInput">
	    <soap:body use="literal"
	      namespace="http://www.service-now.com/incident" />
	  </wsdl:input>
	  <wsdl:output name="IncidentOutput">
	    <soap:body use="literal"
	      namespace="http://www.service-now.com/incident" />
	  </wsdl:output>
	</wsdl:operation>
      </wsdl:binding>
      <wsdl:service name="Incident">
        <wsdl:port binding="tns:IncidentSOAP" name="IncidentSOAP">
          <soap:address location="https://[instance name].service-now.com/script.do?name=SOAPIncidentCreate&SOAP"/>
        </wsdl:port>
      </wsdl:service>
</wsdl:definitions>

Change the instance name in the soap:address element to reflect the instance that is receiving the SOAP invocation

When the web service is invoked, the following SOAP endpoint is defined as a global business rule to receive the call. Create a global business rule with a function named SOAPIncidentCreate.

Note: These API calls changed in the Calgary release:
  • GlideXMLUtil replaces Packages.com.glide.util.XMLUtil
  • GlideXMLDocument replaces Packages.com.glide.util.XMLDocument

The new script object calls apply to the Calgary release and beyond. For releases prior to Calgary, substitute the packages calls as described above. Packages calls are not valid beginning with the Calgary release. For more information, see Scripting API Changes. 

function SOAPIncidentCreate() {
  gs.log("@@@@@@@Calling SOAPIncidentCreate function...");
  var session = gs.getSession();
  session.setInteractive(true);
  session.impersonate("fcce5636d0bba7fe0163c17dc1912b42");

  var xmlutil = new GlideXMLUtil();
  var result = SoapDocument.createElement("m:CreateResponse");
  result.setAttribute("xmlns:m","http://www.service-now.com/incident");

  var xmlDoc = new GlideXMLDocument();
  xmlDoc.parse(SoapDocumentXML);

  var alert_details = getXpathText(xmlDoc,"//alert_details");
  var alert_title = getXpathText(xmlDoc,"//alert_title");
  var alert_category = getXpathText(xmlDoc,"//alert_category");
  var assigned_to = getXpathText(xmlDoc,"//technician_assigned");
  var alert_id = getXpathText(xmlDoc,"//alert_id");
  var alert_severity = getXpathText(xmlDoc,"//alert_severity");
  var alert_status = getXpathText(xmlDoc,"//alert_status");
  var ticket_priority = getXpathText(xmlDoc,"//ticket_priority");
  var device_id = getXpathText(xmlDoc,"//device_id");
  var device_name = getXpathText(xmlDoc,"//device_name");
  var site_id = getXpathText(xmlDoc,"//site_id");
  var site_name = getXpathText(xmlDoc,"//site_name");
  var site_contact = getXpathText(xmlDoc,"//site_contact");
  var site_phone = getXpathText(xmlDoc,"//site_phone");
  var site_email = getXpathText(xmlDoc,"//site_email");
  var alert_url = getXpathText(xmlDoc,"//alert_url");
  var alert_guid = getXpathText(xmlDoc,"//alert_guid");

  var gr = new GlideRecord("incident");

  gr.addQuery("u_lp_alert_id", alert_id);
  gr.query();
  if (!gr.next()) {
    gs.log("*****SCRIPT***** alert_id: " + alert_id);
    gr.initialize();
    gr.category = "Monitoring";
    gr.subcategory = "Device";
    gr.u_call_source = "LPI";
    gr.short_description = alert_title;
    var sComments = new String(alert_details);
    gr.u_lp_alert_id = alert_id;
    gr.u_lp_alert_guid = alert_guid;
    gr.u_lp_site_guid = site_id;
    gr.severity = alert_severity;
    gr.incident_state = "1";
    gr.priority = ticket_priority;
    sComments = '[code]' + sComments + '\n\n<a href="https://207.35.201.27/SC' + alert_url + 
                '" target="new">https://207.35.201.27/SC' + alert_url + '</a>' + 
                '\n\ndevice_name: ' + device_name + '\ndevice_id: ' + device_id + 
                '\nsite_name ' + site_name + '\nsite_id: ' + site_id + 
                '\nsite_contact: ' + site_contact + '\nsite_phone: ' + site_phone + 
                '\nsite_email: ' + site_email + '\nassigned_to: ' + assigned_to + 
                '\nalert_guid: ' + alert_guid + '[/code]';
    gr.comments = sComments;

    //get company record -- if doesn't exist, then add.
    var co_id;
    var grcpy = new GlideRecord("core_company");
    grcpy.addQuery("name", site_name);
    grcpy.query();
    if (grcpy.next()) {
      gs.log("Found company: " + site_id);
      gr.company = grcpy.sys_id;
      co_id = grcpy.sys_id;
    } else {
      gs.log("Adding company: " + site_id);
      grcpy.name = site_name;
      grcpy.customer = "true";
      co_id = grcpy.insert();
      gr.company = co_id;
    }

    //get ci record -- if doesn't exist, then add.
    var ci_id;
    var grci = new GlideRecord("cmdb_ci");
    grci.addQuery("name", device_name);
    grci.query();
    if (grci.next()) {
      gr.cmdb_ci = grci.sys_id;
      ci_id = grci.sys_id;
    } else {
      grci.name = device_name;
      grci.sys_class_name = "cmdb_ci_computer";
      grci.company = co_id;
      ci_id = grci.insert();
      gr.cmdb_ci = ci_id;
    }

    var contactArray = site_contact.split(" ");
    var s_user;
    var gruser = new GlideRecord("sys_user");
    gruser.addQuery("name", site_contact);
    gruser.query();
    if (gruser.next()) {
      gr.caller_id = gruser.sys_id;
      s_user = gruser.sys_id;
    } else {
      gruser.first_name = contactArray[0];
      gruser.last_name = contactArray[1];
      gruser.company = co_id;
      gruser.phone = site_phone;
      gruser.email = String(site_email);
      s_user = gruser.insert();
      gr.caller_id = s_user;
    }

    if (!grcpy.u_monitoring_group.isNil()) {
      gr.assignment_group = grcpy.u_monitoring_group;
    } else {
      gr.assignment_group.setDisplayValue("DIS Systems");
    }

    var sysid = gr.insert();
    gr.get(sysid);

    createElement("output", gr.number, result);

    gs.log("End of SOAPIncidentCreate function...");

    return result;
  }
}


function getXpathText(doc, xpath) {
  var nl = doc.selectNodes(xpath);
  return GlideXMLUtil.getAllText(nl.item(0));
}

function createElement(name, value, result) {
  var el = SoapDocument.createElement(name);
  var val = SoapDocument.createTextNode(value);
  el.appendChild(val);
  result.appendChild(el);
}

This business rule inserts a new incident, and creates a new company, ci and contact person if they are not already found.

2.2 Outbound Updates from Incident to LPI

Request the following credentials from LPI to use the ProcessTTicketRequest (API version 2.0) web service.

  1. UserName
  2. Password
  3. VarDomain
  4. originID
  5. site GUID

To update LPI alerts from ServiceNow, we use SOAP Envelope to initiate a call to their Level Platforms ProcessTTicketRequest web service to update the alert.

The ProcessTTicketRequest SOAP Envelope looks like this 

<SOAP-ENV:Envelope SOAP-ENV:encodingStyle="http://schemas.xmlsoap.org/soap/encoding/">
  <SOAP-ENV:Header>
    <AuthHeader>
      <UserName>XXX</UserName>
      <Password>8XXEE66E6BXXXXXXXXXX69FXXX54</Password>
      <VarDomain>XXX</VarDomain>
      <ApiVersion>2.0</ApiVersion>
    </AuthHeader>
  </SOAP-ENV:Header>
  <SOAP-ENV:Body>
    <ProcessTTicketRequest>
      <actiondoc>
	<nocsvc>
          <callee originid="29EXXX340D253CXXXE59478XXX9B5EB9" siteguid="1XXX162f-0XXX-4XXX-80XX-7XXXXdaXXXee"/>
          <req action="set" selection="updateticket"/>
	    <params>
              <param dir="in" nm="ticketid" val="3180"/>
              <param dir="in" nm="trbstatus" val="3"/>
            </params>
        </nocsvc>
      </actiondoc>
    </ProcessTTicketRequest>
  </SOAP-ENV:Body>
</SOAP-ENV:Envelope>

Create the business rule with conditions to observe desirable changes to incident to be triggered and send the SOAP Envelope through the ECC Queue. An example of this business rule follows 

var soap = new SOAPEnvelope();

var co_domain = current.company.parent.u_company_code;
var site_guid = current.u_lp_site_guid;

var header = soap.createHeaderElement("AuthHeader", "", "xmlns", "http://www.levelplatforms.com/nocsupportservices/2.0/");
var user_name = soap.createElement(header, "UserName", "<request user from LPI>");
var password = soap.createElement(header, "Password", "<request password from LPI>");
var varDomain = soap.createElement(header, "VarDomain", co_domain);
var ApiVersion = soap.createElement(header, "ApiVersion", "2.0");

var sendComments = "";
if (current.comments.changes()) {
  sendComments = sendComments + current.comments.getJournalEntry(1);
}
if (current.work_notes.changes()) {
  sendComments = sendComments + current.work_notes.getJournalEntry(1);
}

//Set body info.
var body = soap.createBodyElement("ProcessTTicketRequest");
soap.setAttribute(body, "xmlns","http://www.levelplatforms.com/nocsupportservices/2.0/");

var actdoc = soap.createElement(body, "actiondoc", "");

var nocsvc = soap.createElement(actdoc, "nocsvc", "");
soap.setAttribute(nocsvc, "xmlns", "");

var callee = soap.createElement(nocsvc, "callee", "");
soap.setAttribute(callee, "originid", "<request origin ID from LPI>");
soap.setAttribute(callee, "siteguid", site_guid);

var req = soap.createElement(nocsvc, "req", "");
soap.setAttribute(req, "action", "set");
soap.setAttribute(req, "selection", "updateticket");

var params1 = soap.createElement(nocsvc, "params", "");
var param1 = soap.createElement(params1, "param", "");
soap.setAttribute(param1, "dir", "in");
soap.setAttribute(param1, "nm", "ticketid");
soap.setAttribute(param1, "val", current.u_lp_alert_id);

var param2 = soap.createElement(params1, "param", "");
soap.setAttribute(param2, "dir", "in");
soap.setAttribute(param2, "nm", "trbstatus");
soap.setAttribute(param2, "val", "3");

soap.invokeService("https://lpi.mspalerts.com/scmessaging/scnoc.asmx", 
"http://www.levelplatforms.com/nocsupportservices/2.0/ProcessTTicketRequest",
                     "Level Platforms Integration");

The resulting response from Level Platforms looks like this (in the ECC Queue as an input record) 

<soap:Envelope>
  <soap:Header>
    <wsa:Action>http://www.levelplatforms.com/nocsupportservices/2.0/ProcessTTicketRequestResponse</wsa:Action>
    <wsa:MessageID>urn:uuid:d048b268-5b7d-45e9-b206-abeaa0eadd72</wsa:MessageID>
    <wsa:RelatesTo>urn:uuid:389872b0-aa69-4d46-91bd-eeaa2f02d7f5</wsa:RelatesTo>
    <wsa:To>http://schemas.xmlsoap.org/ws/2004/08/addressing/role/anonymous
    </wsa:To>
    <wsse:Security>
      <wsu:Timestamp wsu:Id="Timestamp-04d7b7e1-fe86-4b7c-8e2e-f4d289ef6dc3">
        <wsu:Created>2008-11-05T23:02:23Z</wsu:Created>
        <wsu:Expires>2008-11-06T23:02:23Z</wsu:Expires>
      </wsu:Timestamp>
    </wsse:Security>
  </soap:Header>
  <soap:Body>
    <ProcessTTicketRequestResponse>
      <ProcessTTicketRequestResult>
        <nocsvc>
          <status>OK</status>
          <response>Ticket updated successfully.</response>
          <wsdetails selection="updateticket"
           wsdlurl="https://lpi.mspalerts.com/scmessaging/scnoc.asmx"
           wsmethod="ProcessTTicketRequest()"/>
        </nocsvc>
      </ProcessTTicketRequestResult>
    </ProcessTTicketRequestResponse>
  </soap:Body>
</soap:Envelope>

You may optionally process this response to acknowledge the alert update.
