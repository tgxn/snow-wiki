Name: Calculate Percent of Critical Fields

Type:

Table:

Description: Calculation that gives the % of critical fields (in this example, with an attribute of "tiaa_critical") that have been filled out.

Parameters:

Script: 

var totalCritical = 0;
var filledCritical = 0;
var fields = current.getFields();
gs.print(fields);

for (var num = 0; num < fields.size(); num++) { 
  gs.print("----------------RUNNING ARRAY VALUE " + num);
  var ed = fields.get(num).getED();
  if(ed.hasAttribute("tiaa_critical")) {
    gs.print("----------------CRITICAL FIELD FOUND--------------------");
    totalCritical ++;
    if (!fields.get(num).isNil()) {
      filledCritical ++;
    }
  }
}

var answer = 0;
gs.print("-----------------TOTAL - " + totalCritical);
gs.print("-----------------FILLED - " + filledCritical);

if (filledCritical > 0 && totalCritical > 0) { 
  var pcnt = (filledCritical/totalCritical) * 100;
  answer = pcnt.toFixed(2);   
}

answer;

Name: Change File Names in Directory to Lowercase

Type:

Table:

Description: Formats the filenames in a given directory to lowercase.

Parameters:

Script: 

var DIRECTORY = 'c:/temp/Knowledge/Solutions/images/';
doRename(DIRECTORY);
function doRename(directory) {
  var file = new java.io.File(directory);
  if (!file.exists()) {
      gs.addInfoMessage('Could not execute import from directory ' + directory + ' as it does not exist.');
      return;
  }
  if (!file.isDirectory()) {
      gs.addInfoMessage('Could not execute import from directory ' + directory + ' as it is not a directory.');
      return;
  }
  var contents = file.list();
  for (var x =0; x&lt; contents.length; x++) {
      var fileName = contents[x];      
      gs.print('Processing file ' + fileName);
      var lowerName = fileName.toLowerCase();
      var file = new java.io.File(directory + fileName);
      var newFile = new java.io.File(directory + lowerName);
      var success = file.renameTo(newFile);
      if (success) {
          gs.print('SUCCESSFULLY renamed ' + file + ' to ' + newFile);
      } else {
          gs.print('FAILED to rename ' + file + ' to ' + newFile);
      }    
  }
}

Name: Convert Time to Military Time

Type:

Table:

Description: The following script splits a date/time string in the format "MM/DD/YYYY H:mm:SS AM" to military time in the format "MM/DD/YYYY H:mm:SS" and stores that value in a variable "dateTime".

Parameters:

Script: 
var dateTime="11/30/2009 1:00:00 PM";
//Split the dateTime to get the time
var date = dateTime.split(" ")[0];
var time = dateTime.split(" ")[1];
var ampm = dateTime.split(" ")[2];
 
//If 'PM' found then convert
if(ampm.toUpperCase() == 'PM'){
  //Split the time to get the hours
  var hours = parseInt(time.split(":")[0]);
  var minutes = time.split(":")[1];
  var seconds = time.split(":")[2];
 
  //Add 12 hours
  hours = (hours + 12).toString();
 
  //Reassemble the time
  time = hours + ":" + minutes + ":" + seconds;
  //Reassemble the dateTime
  dateTime = date + " " + time;
}
Note: This article applies to Fuji. For more current information, see Delete All Records from a Table at http://docs.servicenow.com

Name: Remove All Data from a Table

Type: Background Script.

Table: Specified in the function parameter.

Description: This script removes all data from the table specified in the function parameter. It will also delete referenced data as well. For details on the used function look at GlideRecord documentation Use extreme caution when running this script. You can also delete all data from a table through the user interface.

To run this script:

  1. Navigate to System Definition > Scripts - Background.
  2. Copy the sample script into the script field.
  3. Change table_name in the script to the name of the table you want to delete all records from.

Parameters: The name of the table to delete all data from.

Script: 
doit("table_name");
//for example doit("cmdb_ci");

function doit(table) {
  var gr = new GlideRecord(table);
  gr.query();
  gr.deleteMultiple();
}

Name: Search for Text in Business Rules

Type:

Table:

Description: This script locates business rules with a given text and returns them in a list. Note: this functionality can be more easily achieved with the Text Search Plugin.

Parameters:

Script: 

findit('string you are searching for');

function findit(str) { 

 var scr = "";
var gr1 = new GlideRecord('sys_script');
gr1.query();
while (gr1.next()) {
  scr = gr1.script.toString();
  if (scr.indexOf(str) &gt; -1) {
    gs.addInfoMessage(gr1.name);
  }
}

}

Name: Update Install Count Based on CI Status

Type: Business Rule

Table:

Description: 'Before' business rule run on cmdb_ci_computer updates that adds (or subtracts) the install count for all of a computers' software based on its status. Make sure to set the condition: 

(current.operational_status == '1') || (current.operational_status == '2')

Parameters:

Script: 

if(current.operational_status.changes()){
gs.print("++++Operational status has changed++++");
 if (current.operational_status == '2'){
   gs.print("++++New status is Non-Operational++++");
   var appList= new GlideRecord("cmdb_software_instance");
   appList.addQuery("installed_on", "=", current.sys_id);
   appList.query();
      sw.install_count --;
       sw.update();
       gs.print("New count - " + sw.install_count);
     }
   }
}
if (current.operational_status == '1'){
  gs.print("++++New status is Operational++++");
  var appList2 = new GlideRecord("cmdb_software_instance");
  appList2.addQuery("installed_on", "=", current.sys_id);
  appList2.query();
  while(appList2.next()) {
    gs.print(appList2.getDisplayValue() + " found");
    var sw2 = new GlideRecord("cmdb_ci_spkg");
    sw2.addQuery('sys_id', "=", appList2.software);
    sw2.query();
    while(sw2.next()) {
      gs.print("AND matching " + sw2.getDisplayValue() + " found...increasing count");
      gs.print("Original count - " + sw2.install_count);
      sw2.install_count ++;
      sw2.update();
      gs.print("New count - " + sw2.install_count);      
    }
  }
}
